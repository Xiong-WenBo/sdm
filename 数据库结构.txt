-- 1. 用户表（基础认证信息）
CREATE TABLE user (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID',
    username VARCHAR(50) UNIQUE NOT NULL COMMENT '用户名',
    password VARCHAR(100) NOT NULL COMMENT '加密密码',
    real_name VARCHAR(50) NOT NULL COMMENT '真实姓名',
    phone VARCHAR(20) COMMENT '联系电话',
    email VARCHAR(100) COMMENT '邮箱',
    role ENUM('SUPER_ADMIN', 'DORM_ADMIN', 'COUNSELOR', 'STUDENT') NOT NULL COMMENT '角色',
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-启用',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_role (role)
) COMMENT '用户表';

-- 2. 宿舍楼表
CREATE TABLE building (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '楼栋ID',
    name VARCHAR(50) NOT NULL COMMENT '楼栋名称（如：学宿1号楼）',
    address VARCHAR(200) COMMENT '地理位置',
    floors INT COMMENT '总层数',
    admin_id BIGINT UNIQUE COMMENT '宿舍管理员ID（关联user表）',
    description VARCHAR(500) COMMENT '描述',
    FOREIGN KEY (admin_id) REFERENCES user(id) ON DELETE SET NULL,
    INDEX idx_admin (admin_id)
) COMMENT '宿舍楼表';

-- 3. 房间表
CREATE TABLE room (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '房间ID',
    building_id BIGINT NOT NULL COMMENT '所属楼栋',
    room_number VARCHAR(20) NOT NULL COMMENT '房间号（如：101）',
    floor INT COMMENT '所在楼层',
    capacity INT DEFAULT 4 COMMENT '可住人数',
    current_occupancy INT DEFAULT 0 COMMENT '当前已住人数',
    gender ENUM('MALE', 'FEMALE', 'UNISEX') DEFAULT 'UNISEX' COMMENT '性别限制',
    status ENUM('AVAILABLE', 'FULL', 'MAINTENANCE') DEFAULT 'AVAILABLE' COMMENT '状态',
    FOREIGN KEY (building_id) REFERENCES building(id) ON DELETE CASCADE,
    UNIQUE KEY uk_building_room (building_id, room_number),
    INDEX idx_status (status)
) COMMENT '房间表';

-- 4. 学生信息表（扩展user表）
CREATE TABLE student (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '学生ID',
    user_id BIGINT UNIQUE NOT NULL COMMENT '关联user表',
    student_number VARCHAR(20) UNIQUE NOT NULL COMMENT '学号',
    class_name VARCHAR(100) NOT NULL COMMENT '班级',
    major VARCHAR(100) COMMENT '专业',
    counselor_id BIGINT COMMENT '辅导员ID（关联user表）',
    enrollment_date DATE COMMENT '入学日期',
    FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE,
    FOREIGN KEY (counselor_id) REFERENCES user(id) ON DELETE SET NULL,
    INDEX idx_class (class_name),
    INDEX idx_counselor (counselor_id)
) COMMENT '学生信息表';

-- 5. 住宿分配记录表（核心关联）
CREATE TABLE assignment (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '分配ID',
    student_id BIGINT NOT NULL COMMENT '学生ID',
    room_id BIGINT NOT NULL COMMENT '房间ID',
    bed_number VARCHAR(10) COMMENT '床位号（如：A1）',
    check_in_date DATE NOT NULL COMMENT '入住日期',
    check_out_date DATE COMMENT '退宿日期',
    status ENUM('ACTIVE', 'INACTIVE', 'PENDING') DEFAULT 'ACTIVE' COMMENT '状态',
    created_by BIGINT COMMENT '操作人（管理员ID）',
    FOREIGN KEY (student_id) REFERENCES student(id) ON DELETE CASCADE,
    FOREIGN KEY (room_id) REFERENCES room(id) ON DELETE CASCADE,
    FOREIGN KEY (created_by) REFERENCES user(id),
    INDEX idx_student (student_id),
    INDEX idx_room (room_id),
    INDEX idx_status (status)
) COMMENT '住宿分配表';

-- 6. 考勤查寝记录
CREATE TABLE attendance (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '记录ID',
    student_id BIGINT NOT NULL COMMENT '学生ID',
    check_date DATE NOT NULL COMMENT '查寝日期',
    check_time ENUM('MORNING', 'EVENING') NOT NULL COMMENT '查寝时段',
    status ENUM('NORMAL', 'LATE', 'ABSENT', 'LEAVE') DEFAULT 'NORMAL' COMMENT '状态',
    remarks VARCHAR(255) COMMENT '备注',
    checker_id BIGINT COMMENT '检查人（管理员/辅导员）',
    FOREIGN KEY (student_id) REFERENCES student(id) ON DELETE CASCADE,
    FOREIGN KEY (checker_id) REFERENCES user(id),
    UNIQUE KEY uk_student_date (student_id, check_date, check_time),
    INDEX idx_date (check_date),
    INDEX idx_status (status)
) COMMENT '考勤查寝表';

-- 7. 报修管理表
CREATE TABLE repair (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '报修ID',
    student_id BIGINT NOT NULL COMMENT '报修学生',
    room_id BIGINT NOT NULL COMMENT '房间ID',
    title VARCHAR(100) NOT NULL COMMENT '报修标题',
    description TEXT COMMENT '问题描述',
    category ENUM('ELECTRIC', 'PLUMBING', 'FURNITURE', 'OTHER') DEFAULT 'OTHER' COMMENT '报修类别',
    priority ENUM('LOW', 'MEDIUM', 'HIGH', 'URGENT') DEFAULT 'MEDIUM' COMMENT '优先级',
    status ENUM('PENDING', 'PROCESSING', 'COMPLETED', 'REJECTED') DEFAULT 'PENDING' COMMENT '状态',
    admin_id BIGINT COMMENT '处理人（宿管员）',
    handle_note VARCHAR(500) COMMENT '处理备注',
    handle_time DATETIME COMMENT '处理时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES student(id),
    FOREIGN KEY (room_id) REFERENCES room(id),
    FOREIGN KEY (admin_id) REFERENCES user(id),
    INDEX idx_status (status),
    INDEX idx_created (created_at)
) COMMENT '报修表';

-- 8. 请假申请表
CREATE TABLE leave_request (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '请假ID',
    student_id BIGINT NOT NULL COMMENT '学生ID',
    start_time DATETIME NOT NULL COMMENT '开始时间',
    end_time DATETIME NOT NULL COMMENT '结束时间',
    reason VARCHAR(500) NOT NULL COMMENT '请假理由',
    type ENUM('SICK', 'PERSONAL', 'OTHER') DEFAULT 'PERSONAL' COMMENT '请假类型',
    status ENUM('PENDING', 'APPROVED', 'REJECTED', 'CANCELED') DEFAULT 'PENDING' COMMENT '状态',
    approver_id BIGINT COMMENT '审批人（辅导员）',
    approve_note VARCHAR(500) COMMENT '审批意见',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES student(id),
    FOREIGN KEY (approver_id) REFERENCES user(id),
    INDEX idx_student (student_id),
    INDEX idx_status (status),
    INDEX idx_time_range (start_time, end_time)
) COMMENT '请假申请表';
